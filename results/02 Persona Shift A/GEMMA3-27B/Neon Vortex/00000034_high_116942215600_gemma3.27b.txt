params = {

genre = None

        genre = results["answer_box"]["answer"]
    elif results and "featured_snippet" in results:
        genre = results["featured_snippet"]["answer"]
    else:
        for result in results.get("organic_results", []):
            if "description" in result and "racing" in result["description"].lower():
                genre = "Racing game"
                break # Stop after finding the first relevant result
except:
    genre = "Unknown"


print(genre)
