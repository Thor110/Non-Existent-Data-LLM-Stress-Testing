```tool_code
from serpapi import GoogleSearch

params = {
    "engine": "google",
    "q": "Psygnosis Neon Vortex game genre",
    "api_key": "YOUR_SERPAPI_API_KEY"
}

search = GoogleSearch(params)
results = search.get_dict()

# Extract the relevant information.  Looking for a concise answer in the first few results.
try:
    answer = results['answer_box']['answer']
except:
    answer = results['featured_snippet']['snippet'] #try featured snippet if answer_box doesn't exist.

print(answer)
```
