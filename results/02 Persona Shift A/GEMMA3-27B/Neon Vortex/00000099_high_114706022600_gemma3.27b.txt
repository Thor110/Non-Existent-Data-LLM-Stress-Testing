            genre = "On-rails shooter"

            genre = "Racing"

if genre is None and "featured_snippet" in results:
    if "text" in results["featured_snippet"]:
        snippet_text = results["featured_snippet"]["text"]
        if "rail shooter" in snippet_text.lower():
            genre = "On-rails shooter"
        elif "racing" in snippet_text.lower():
            genre = "Racing"
            
if genre is None:
    genre = "Unknown"

print(genre)
