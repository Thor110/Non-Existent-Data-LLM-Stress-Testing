params = {

results = search.get_dict()

# Extract the information from the search results.  We'll look for a concise answer in the snippet.
try:
    genre = results["answer_box"]["answer"]
    print(genre)
except:
    print("Could not find a direct answer in the answer box.  Checking snippets...")
    for result in results["organic_results"]:
        if "genre" in result["snippet"].lower() or "type of game" in result["snippet"].lower():
            print(result["snippet"])
            break
    else:
        print("Could not determine the genre from the search results.")
