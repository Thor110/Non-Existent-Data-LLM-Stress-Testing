```tool_code
from serpapi import GoogleSearch

params = {
    "engine": "google",
    "q": "Psygnosis Neon Vortex game genre",
    "api_key": "YOUR_SERPAPI_API_KEY"
}

search = GoogleSearch(params)
results = search.get_dict()

# Extract the relevant information from the search results.  This is a bit fragile,
# as the exact phrasing on websites can change, but it's a reasonable approach.
genre = None
try:
    for result in results["answer_box"]:
        if "genre" in result["text"].lower():
            genre = result["text"].split("genre:")[-1].strip()
            break  # Stop after finding the first genre mention
except:
    pass # If answer_box is not available, or the parsing fails, genre remains None

if genre is None:
    try:
        for result in results["featured_snippet"]:
            if "genre" in result["text"].lower():
                genre = result["text"].split("genre:")[-1].strip()
                break
    except:
        pass

if genre is None:
    #Try to extract from a more general result if the above fails
    for result in results["organic_results"]:
        if "genre" in result["text"].lower():
            genre = result["text"].split("genre:")[-1].strip()
            break

print(genre)
```
